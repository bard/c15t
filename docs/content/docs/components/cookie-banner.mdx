## Features

- Customizable cookie consent modal
- Accordion-style cookie preference management
- Integration with React Hook Form
- Support for required and optional cookie types
- Customizable UI elements and content

## Usage

```tsx
import { CookieConsentModal, CookieConsentAccept, CookieConsentDecline } from "@/components/ui/cookie-consent-modal"
import { useAnalytics } from "@/lib/analytics"

export function CookieConsent() {
  const analytics = useAnalytics()

  return (
    <CookieConsentModal
      useAnalytics={analytics}
      requiredGdprPurposes={["strictly_necessary"]}
    >
      <h2 className="text-lg font-semibold">We value your privacy</h2>
      <p className="mt-2">
        We use cookies to enhance your browsing experience, serve personalized
        ads or content, and analyze our traffic. By clicking "Accept All", you
        consent to our use of cookies.
      </p>
      <div className="flex justify-end space-x-4 mt-4">
        <CookieConsentDecline>Decline</CookieConsentDecline>
        <CookieConsentAccept>Accept All</CookieConsentAccept>
      </div>
    </CookieConsentModal>
  )
}
```

## Examples

### Custom Styling

You can customize the appearance of the modal by passing className and style props:

```tsx
<CookieConsentModal
  className="bg-secondary text-secondary-foreground"
  style={{ borderRadius: "16px" }}
  useAnalytics={analytics}
  requiredGdprPurposes={["strictly_necessary"]}
>
</CookieConsentModal>
```

### Custom Consent Item Rendering

You can provide a custom rendering function for each consent item:

```tsx
<CookieConsentModal
  useAnalytics={analytics}
  requiredGdprPurposes={["strictly_necessary"]}
  renderConsentItem={({ cookie, control }) => (
    <div key={cookie.name} className="flex items-center justify-between py-2">
      <span>{cookie.name}</span>
      <Switch
        checked={control.watch(cookie.name)}
        onCheckedChange={(checked) => control.setValue(cookie.name, checked)}
      />
    </div>
  )}
>
</CookieConsentModal>
```
